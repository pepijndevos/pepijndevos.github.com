Don't risk your date, get PyMouse right now at its GitHub page! All that it takes to wiggle your pointer with PyMouse is this: 
from pymouse import PyMouse
from time import sleep
from random import randint

m = PyMouse()
w, h = m.screen_size()
while True:
    try:
        m.move(randint(0, w), randint(0, h))
        sleep(20)
    except KeyboardInterrupt:
        print "Goodby!"
        break
